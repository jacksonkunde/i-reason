class DatasetSplitter:
    """
    Splits the dataset into training, validation, and test sets based on hash values and other criteria.
    """

    def __init__(
        self,
        hash_threshold: int = 17,
        max_instance_params: int = 5,
        max_operations: int = 10,
        mod_value: int = 23,
    ):
        """
        Initializes the DatasetSplitter with specified parameters.

        :param hash_threshold: Threshold for splitting based on hash value.
        :param max_instance_params: Maximum number of instance parameters for training data.
        :param max_operations: Maximum number of operations for training data.
        :param mod_value: The modulus value used in hash computations.
        """
        self.hash_threshold = hash_threshold
        self.max_instance_params = max_instance_params
        self.max_operations = max_operations
        self.mod_value = mod_value

    def split(
        self, dataset: List[Dict[str, Any]]
    ) -> Tuple[
        List[Dict[str, Any]],
        List[Dict[str, Any]],
        List[Dict[str, Any]],
        List[Dict[str, Any]],
    ]:
        """
        Splits the dataset into training, validation, test set 1, and test set 2.

        :param dataset: The list of data points to split.
        :return: A tuple containing the training set, validation set, test set 1, and test set 2.
        """
        train_set = []
        val_set = []
        test_set_1 = []
        test_set_2 = []

        for data_point in dataset:
            hash_value = data_point["hash_value_mod_23"]
            instance_params_count = data_point["instance_params_count"]
            operations_count = data_point["operations_count"]

            if (
                hash_value < self.hash_threshold
                and instance_params_count <= self.max_instance_params
                and operations_count <= self.max_operations
            ):
                # Assign to training or validation set
                if random.random() < 0.1:
                    val_set.append(data_point)
                else:
                    train_set.append(data_point)
            elif (
                hash_value >= self.hash_threshold
                and instance_params_count <= self.max_instance_params
                and operations_count <= self.max_operations
            ):
                test_set_1.append(data_point)
            else:
                test_set_2.append(data_point)

        return train_set, val_set, test_set_1, test_set_2
